187:     
188:     # default to no cookie key found:
189:     cookie_key = false
190:     
191:     # gets the cookie array from the request object
192:     cookie_raw = msg.request.cookies
193:     
194:     # checks, if a cookie named 'ses_key' is found
195:     if cookie_raw.has_key?('ses_key')
196:       
197:       # gets just the data itself (discards comment, domain, exipiry etc)
198:       cookie_key = cookie_raw['ses_key'].split(':')[0]
199:       
200:     end
201:     
202:     # if a cookie key is found (non-false), checks if it's valid
203:     if cookie_key
204:       
205:       # checks for validity by looking the key up in @session_cookie_keys
206:       cookie_key_exist = @session_cookie_keys.has_key?( cookie_key )
207:       
208:       # sets the cookie key to false, if it doesn't exist
209:       cookie_key = false unless cookie_key_exist
210:       
211:     end
212:     
213:     # at this point, the cookie key seems valid:
214:     if cookie_key and cookie_key_exist
215:       
216:       # get the session identifier
217:       ses_id = @session_cookie_keys[ cookie_key ]
218:       
219:       # get the last session key from session data
220:       ses_key = @sessions[ses_id][:ses_key]
221:       
222:       # make additional checks on the session validity (expiry etc)
223:       ses_status = check_ses( msg, ses_key )
224:       
225:       # delete the old key:
226:       @session_cookie_keys.delete( cookie_key )
227:       
228:       # status is either true (valid) or false (invalid)
229:       # based on the result of check_ses
230:       if ses_status
231:         
232:         # get a new cookie key
233:         cookie_key = @randgen.get(@config[:cookie_key_multiplier]).join('')
234:         
235:         # map the new cookie key to the old session identifier
236:         @session_cookie_keys[ cookie_key ] = ses_id
237:         
238:         # binds the new cookie key to the old session data
239:         @sessions[ses_id][:cookie_key] = cookie_key
240:         
241:         # Sets the restored_session flag of msg to true
242:         # It signals plugins to re-set data
243:         msg.restored_session = true
244:         
245:         # Sets the new_session flag of msg to false
246:         # It signals plugins to not create new server-side values
247:         msg.new_session = false
248:         
249:         # tells ValueManager to re-send client-side HValue objects
250:         # with data to the client
251:         VALUES.resend_session_values( msg )
252:       
253:       # if the session is not valid, make sure to mark the
254:       # cookie key as invalid (false)
255:       else
256:         cookie_key = false
257:       end
258:     end
259:     
260:     # if the cookie key failed validation in the
261:     # tests above, create a new session instead
262:     unless cookie_key
263:       cookie_key = init_ses( msg )
264:       ses_status = true
265:     end
266:     
267:     # Uses a cookie comment to tell the user what the
268:     # cookie is for, change it to anything valid in the
269:     # configuration.
270:     ses_cookie_comment = @config[:ses_cookie_comment]
271:     
272:     ## mod_rewrite changes the host header to x-forwarded-host:
273:     if msg.request.header.has_key?('x-forwarded-host')
274:       domain = msg.request.header['x-forwarded-host']
275:

292:       ses_cookie_domain  = domain

298:     ses_cookie_max_age = @config[:timeout_secs]

114:   def check_ses( msg, ses_key )
115:     
116:     # first, check if the session key exists (xhr)
117:     if @session_keys.has_key?( ses_key )
118:       
119:       # get the session's id based on its key
120:       ses_id   = @session_keys[ ses_key ]
121:       
122:       # get the session's data based on its id
123:       ses_data = @sessions[ ses_id ]
124:       
125:       # new time-out
126:       ses_data[:timeout] = Time.now.to_i + @config[:timeout_secs]
127:       
128:       ### extra security
129:       # re-generates the ses_key for each xhr
130:       if @config[:disposable_keys]
131:         
132:         # disposes the old (current) ses_key:
133:         @session_keys.delete( ses_key )
134:         
135:         # gets a new ses_key:
136:         ses_key = @randgen.get_one
137:         
138:         # re-maps the session id to the new key
139:         @session_keys[ses_key] = ses_id
140:         
141:         # changes the session key in the session data
142:         ses_data[:ses_key] = ses_key
143:         
144:         # tell the client what its new session key is
145:         msg.reply "HTransporter.ses_id='#{ses_key}';"
146:       end
147:       ### /extra security
148:       
149:       ### Bind the session data and id to the message object
150:       msg.session = ses_data
151:       msg.ses_id  = ses_id
152:       
153:       ## Return success
154:       return true
155:     
156:     ## The session was either faked or expired:
157:     else
158:       ### Tells the client to stop connecting with its session key and reload instead to get a new one.
159:       stop_client_with_message( msg,
160:         @config[:messages][:invalid_session][:title],
161:         @config[:messages][:invalid_session][:descr],
162:         @config[:messages][:invalid_session][:uri]
163:       )
164:       
165:       ## Return failure
166:       return false
167:     end
168:     
169:   end

346:       msg = Message.new( request, response )
347:       
348:       ## The client tells that its ses_key is '0',
349:       ## until the server tells it otherwise.
350:       if ses_key == '0'

352:         # If Broker encounters a '/hello' request, it
353:         # sets cookies to true.
354:         #
355:         # It means that a session should have its cookies
356:         # checked.
357:         #

 45:   def init_ses( msg )
 46:     
 47:     ## Assigns new timeout for the session
 48:     time_now = Time.now.to_i # seconds since epoch
 49:     timeout  = time_now + @config[:timeout_secs]
 50:     
 51:     ## Creates a new session key
 52:     ses_key    = @randgen.get_one
 53:     
 54:     ## Creates a new cookie key
 55:     cookie_key = @randgen.get(@config[:cookie_key_multiplier]).join('')
 56:     
 57:     ## Makes a new database row for the session, returns its id
 58:     ses_id     = new_ses_id( cookie_key, ses_key, timeout )
 59:     
 60:     ### Default session data structure,
 61:     ### Please don't mess with it, unless you know exactly what you are doing.
 62:     ses_data = {
 63:       
 64:       # the time, when the session will time out
 65:       :timeout    =>  timeout,
 66:       
 67:       # session id, used internally
 68:       :ses_id     =>  ses_id,
 69:       
 70:       # session key, used externally (client xhr)
 71:       :ses_key    =>  ses_key,
 72:       
 73:       # session key, used externally (client cookies)
 74:       :cookie_key =>  cookie_key,
 75:       
 76:       # user id, map to your own user management code
 77:       :user_id    =>  0,
 78:       
 79:       # valuemanager data
 80:       :values     => {
 81:         :sync  => [],  # value id's to sync to client
 82:         :check => [],  # value id's to validate in server (from client)
 83:         :by_id => {}   # values by id
 84:       }
 85:     }
 86:     
 87:     # bind the session data to @sessions by its id
 88:     @sessions[ ses_id ] = ses_data
 89:     
 90:     # map the key back to the id
 91:     @session_keys[ ses_key ] = ses_id
 92:     
 93:     # map the ses_id to cookie key
 94:     @session_cookie_keys[ cookie_key ] = ses_id
 95:     
 96:     ### Tell the client what the new key is
 97:     msg.reply "HTransporter.ses_id='#{ses_key}';"
 98:     
 99:     ### Set the session data and id to the message object
100:     msg.session = ses_data
101:     msg.ses_id  = ses_id
102:     
103:     # Flag the session as new, so associated
104:     # plugins know when to create new data
105:     msg.new_session = true
106:     
107:     # Returns the cookie key, so it can be sent in the response header
108:     return cookie_key
109:     
110:   end
