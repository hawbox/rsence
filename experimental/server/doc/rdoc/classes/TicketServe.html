184:   def serve( msg, content, format='PNG', type=:img )
185:     
186:     # gets a new, unique identifier
187:     ticket_id = @randgen.give_one
188:     
189:     # serve image
190:     if type == :img
191:       
192:       # makes sure the format is in uppper case
193:       format.upcase!
194:       
195:       # checks, that the format is a supported image type
196:       unless @content_types.keys.include?( format )
197:         puts "ImgServe.serve: invalid format (#{format.inspect})" if DEBUG_MODE
198:         return '/img/invalid.gif'
199:       end
200:       
201:       # changes the format to GIF for IE6
202:       format = 'GIF' if msg.ie6 and format == 'PNG'
203:       
204:       # stores the image data and meta-data ready to be served
205:       storage_hash = @imgs
206:       storage_arr = [format,0,content,msg.ses_id]
207:       
208:       # return an uri that will respond to the data
209:       uri = "/i/#{ticket_id}.#{format.downcase}"
210:     
211:     # serve file
212:     elsif type == :file
213:       
214:       # re-map content_type and filename meta-data from format
215:       (content_type, filename) = format
216:       
217:       # content size for the header
218:       content_size = content.size.to_s
219:       
220:       storage_hash = @files
221:       storage_arr = [content_type,content_size,content,msg.ses_id,filename]
222:       
223:       uri = "/f/#{ticket_id}"
224:     end
225:     
226:     # makes sure, that the storage array has a sub-array for sessions (to aid session-based cleanup)
227:     storage_hash[:ses_ids][msg.ses_id] = [] unless @imgs[:ses_ids].has_key?(msg.ses_id)
228:     
229:     # adds the id to the session-specific storage
230:     storage_hash[:ses_ids][msg.ses_id].push( ticket_id )
231:     
232:     storage_hash[:by_id][ticket_id] = storage_arr
233:     
234:     return uri
235:   end
