scan()

Loads pluginlicotions from the list of plugin directories.

[Source]

     # File lib/plugins/pluginmanager.rb, line 106
106:   def scan
107:     
108:     ## Reset soapserve
109:     deinit_soapserve
110:     init_soapserve
111:     
112:     # loop through all plugin-mainlevel directories
113:     @dirs.each do |plugin_dir|
114:       
115:       # makes sure the plugin dir is a dir
116:       # (there might be files like readme.txt or licence.txt)
117:       if(FileTest.directory?(plugin_dir))
118:         
119:         # goes through all plugins in the
120:         # plugin dir in alphabetical order:
121:         Dir.new(plugin_dir).sort.each do |plugin_name|
122:           
123:           # skip, if the plugin starts with a dot, like '.svn', '.' and '..'
124:           next if(plugin_name =~ /^\./)
125:           
126:           # sets the plugin path
127:           @@curr_plugin_path = File.join(plugin_dir,plugin_name)
128:           
129:           # checks that the plugin is a dir
130:           next unless(FileTest.directory?(@@curr_plugin_path))
131:           
132:           # expects to find a 'plugin_dir/plugin_name/plugin_name.rb' file
133:           filename = File.join(@@curr_plugin_path,"#{plugin_name}.rb")
134:           next unless File.exist?(filename)
135:           
136:           # if the plugin contains a 'disabled' flag-file, it skips to the next
137:           next if File.exist?(File.join(@@curr_plugin_path,'disabled'))
138:           
139:           # Create a new, anonymous module as the plugin namespace.
140:           plugin_module = Module.new
141:           
142:           ##### Evaluates the plugin as a string in an anonymous module
143:           begin
144:             plugin_as_string = IO.readlines( filename ).join
145:             plugin_module.module_eval( plugin_as_string )
146:             
147:           ###### If initialization fails, print a nice stack trace
148:           rescue => e
149:             puts "=="*40 if $DEBUG_MODE
150:             puts "WARN: Plugin #{plugin_name} failed to initialize."
151:             if $DEBUG_MODE
152:               puts "--"*40
153:               puts e.message.gsub('(eval):',%{#{File.join(@@curr_plugin_path,"#{plugin_name}.rb")}:})
154:               puts "  #{e.backtrace.join("\n  ")}"
155:               puts "=="*40
156:             end
157:           end
158:           
159:         end
160:         
161:       end
162:       
163:     end
164:     
165:     open()
166:     
167:     @@curr_plugin_path = nil
168:     
169:   end
