201:     
202:     # default to no cookie key found:
203:     cookie_key = false
204:     
205:     # gets the cookie array from the request object
206:     cookie_raw = msg.request.cookies
207:     
208:     # checks, if a cookie named 'ses_key' is found
209:     if cookie_raw.has_key?('ses_key')
210:       
211:       # gets just the data itself (discards comment, domain, exipiry etc)
212:       cookie_key = cookie_raw['ses_key'].split(':')[0]
213:       
214:     end
215:     
216:     # if a cookie key is found (non-false), checks if it's valid
217:     if cookie_key
218:       
219:       # checks for validity by looking the key up in @session_cookie_keys
220:       cookie_key_exist = @session_cookie_keys.has_key?( cookie_key )
221:       
222:       # sets the cookie key to false, if it doesn't exist
223:       cookie_key = false unless cookie_key_exist
224:       
225:     end
226:     
227:     # at this point, the cookie key seems valid:
228:     if cookie_key and cookie_key_exist
229:       
230:       # get the session identifier
231:       ses_id = @session_cookie_keys[ cookie_key ]
232:       
233:       # get the last session key from session data
234:       ses_key = @sessions[ses_id][:ses_key]
235:       
236:       # make additional checks on the session validity (expiry etc)
237:       ses_status = check_ses( msg, ses_key )
238:       
239:       # delete the old key:
240:       @session_cookie_keys.delete( cookie_key )
241:       
242:       # status is either true (valid) or false (invalid)
243:       # based on the result of check_ses
244:       if ses_status
245:         
246:         # get a new cookie key
247:         cookie_key = @randgen.get(@config[:cookie_key_multiplier]).join('')
248:         
249:         # map the new cookie key to the old session identifier
250:         @session_cookie_keys[ cookie_key ] = ses_id
251:         
252:         # binds the new cookie key to the old session data
253:         @sessions[ses_id][:cookie_key] = cookie_key
254:         
255:         # Sets the restored_session flag of msg to true
256:         # It signals plugins to re-set data
257:         msg.restored_session = true
258:         
259:         # Sets the new_session flag of msg to false
260:         # It signals plugins to not create new server-side values
261:         msg.new_session = false
262:         
263:         # tells ValueManager to re-send client-side HValue objects
264:         # with data to the client
265:         $VALUES.resend_session_values( msg )
266:       
267:       # if the session is not valid, make sure to mark the
268:       # cookie key as invalid (false)
269:       else
270:         cookie_key = false
271:       end
272:     end
273:     
274:     # if the cookie key failed validation in the
275:     # tests above, create a new session instead
276:     unless cookie_key
277:       cookie_key = init_ses( msg )
278:       ses_status = true
279:     end
280:     
281:     # Uses a cookie comment to tell the user what the
282:     # cookie is for, change it to anything valid in the
283:     # configuration.
284:     ses_cookie_comment = @config[:ses_cookie_comment]
285:     
286:     ## mod_rewrite changes the host header to x-forwarded-host:
287:     if msg.request.header.has_key?('x-forwarded-host')
288:       domain = msg.request.header['x-forwarded-host']
289:       
290:     ## direct access just uses host (at least mongrel
291:     ## does mod_rewrite header translation):

294:     end
295:     
296:     ## if the host address is a real domain

298:     ## but not an ip-address, prepend it with
299:     ## a dotto accept wildcards (useful for
300:     ## dns-load-balanced server configurations)
301:     if not @ipv4_reg.match(domain) and domain.include?('.')
302:       ses_cookie_domain  = ".#{domain}"
303:     
304:     ## Otherwise, use the domain as-is
305:     else
306:       ses_cookie_domain  = domain
307:     end
308:     
309:     ## uses the timeout to declare the max age
310:     ## of the cookie, allows the browser to delete
311:     ## it, when it expires.
312:     ses_cookie_max_age = @config[:timeout_secs]
313:     
314:     ## Only match the handshaking address of himle,
315:     ## prevents unneccessary cookie-juggling in xhr's
316:     ses_cookie_path    = '/hello'

320:     ses_cookie_str = [

129:   def check_ses( msg, ses_key )
130:     
131:     # first, check if the session key exists (xhr)
132:     if @session_keys.has_key?( ses_key )
133:       
134:       # get the session's id based on its key
135:       ses_id   = @session_keys[ ses_key ]
136:       
137:       # get the session's data based on its id
138:       ses_data = @sessions[ ses_id ]
139:       
140:       # new time-out
141:       ses_data[:timeout] = Time.now.to_i + @config[:timeout_secs]
142:       
143:       ### extra security
144:       # re-generates the ses_key for each xhr
145:       if @config[:disposable_keys]
146:         
147:         # disposes the old (current) ses_key:
148:         @session_keys.delete( ses_key )
149:         
150:         # gets a new ses_key:
151:         ses_key = @randgen.get_one
152:         
153:         # re-maps the session id to the new key
154:         @session_keys[ses_key] = ses_id
155:         
156:         # changes the session key in the session data
157:         ses_data[:ses_key] = ses_key
158:         
159:         # tell the client what its new session key is
160:         msg.ses_key = ses_key
161:       end
162:       ### /extra security
163:       
164:       ### Bind the session data and id to the message object
165:       msg.session = ses_data
166:       msg.ses_id  = ses_id
167:       
168:       ## Return success
169:       return true
170:     
171:     ## The session was either faked or expired:
172:     else
173:       ### Tells the client to stop connecting with its session key and reload instead to get a new one.
174:       stop_client_with_message( msg,
175:         @config[:messages][:invalid_session][:title],
176:         @config[:messages][:invalid_session][:descr],
177:         @config[:messages][:invalid_session][:uri]
178:       )
179:       
180:       ## Return failure
181:       return false
182:     end
183:     
184:   end

 60:   def init_ses( msg )
 61:     
 62:     ## Assigns new timeout for the session
 63:     time_now = Time.now.to_i # seconds since epoch
 64:     timeout  = time_now + @config[:timeout_secs]
 65:     
 66:     ## Creates a new session key
 67:     ses_key    = @randgen.get_one
 68:     
 69:     ## Creates a new cookie key
 70:     cookie_key = @randgen.get(@config[:cookie_key_multiplier]).join('')
 71:     
 72:     ## Makes a new database row for the session, returns its id
 73:     ses_id     = new_ses_id( cookie_key, ses_key, timeout )
 74:     
 75:     ### Default session data structure,
 76:     ### Please don't mess with it, unless you know exactly what you are doing.
 77:     ses_data = {
 78:       
 79:       # the time, when the session will time out
 80:       :timeout    =>  timeout,
 81:       
 82:       # session id, used internally
 83:       :ses_id     =>  ses_id,
 84:       
 85:       # session key, used externally (client xhr)
 86:       :ses_key    =>  ses_key,
 87:       
 88:       # session key, used externally (client cookies)
 89:       :cookie_key =>  cookie_key,
 90:       
 91:       # user id, map to your own user management code
 92:       :user_id    =>  0,
 93:       
 94:       # valuemanager data
 95:       :values     => {
 96:         :sync  => [],  # value id's to sync to client
 97:         :check => [],  # value id's to validate in server (from client)
 98:         :by_id => {}   # values by id
 99:       }
100:     }
101:     
102:     # bind the session data to @sessions by its id
103:     @sessions[ ses_id ] = ses_data
104:     
105:     # map the key back to the id
106:     @session_keys[ ses_key ] = ses_id
107:     
108:     # map the ses_id to cookie key
109:     @session_cookie_keys[ cookie_key ] = ses_id
110:     
111:     ### Tell the client what the new key is
112:     msg.reply "HTransporter.ses_id='#{ses_key}';"
113:     
114:     ### Set the session data and id to the message object
115:     msg.session = ses_data
116:     msg.ses_id  = ses_id
117:     
118:     # Flag the session as new, so associated
119:     # plugins know when to create new data
120:     msg.new_session = true
121:     
122:     # Returns the cookie key, so it can be sent in the response header
123:     return cookie_key
