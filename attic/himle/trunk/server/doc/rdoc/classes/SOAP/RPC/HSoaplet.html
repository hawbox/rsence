process(request, response)

POST-Request goes here:

[Source]

     # File lib/http/soap/hsoaplet.rb, line 55
 55:   def process(request, response)
 56:     
 57:     # Set up some default response values:
 58:     response_status = 500
 59:     content_type    = 'text/xml; charset=UTF-8'
 60:     response_body   = 'Internal Server Errror'
 61:     
 62:     ### Ensures the request method is POST
 63:     if request.env['REQUEST_METHOD'] != 'POST'
 64:       
 65:       response_status = 405
 66:       response['Allow'] = 'POST'
 67:       response_body = 'Method Not Allowed'
 68:       
 69:     else
 70:       
 71:       # The request method is POST
 72:       begin
 73:         
 74:         # The ConnectionData instance keeps copies of the request data and SOAP-related variables
 75:         conn_data = ::SOAP::StreamHandler::ConnectionData.new
 76:         
 77:         # Copies the request body, content type and soap action data to conn_data
 78:         setup_req( conn_data, request )
 79:         
 80:         # Process conn_data with the RPC Router
 81:         conn_data = @router.route( conn_data )
 82:         
 83:         # Sets the status code depending on the success status:
 84:         response_status = conn_data.is_fault ? 500 : 200
 85:         
 86:       rescue Exception => e
 87:         
 88:         # Get a nice error response message:
 89:         conn_data = @router.create_fault_response(e)
 90:         
 91:       end
 92:       
 93:       # Get the content type:
 94:       content_type = conn_data.send_contenttype || content_type
 95:       
 96:       # Render the response body:
 97:       response_body = conn_data.send_string
 98:     end
 99:     
100:     # Finalize the response:
101:     response.status          = response_status
102:     response['Content-Type'] = content_type
103:     response.body            = response_body
104:     
105:   end
