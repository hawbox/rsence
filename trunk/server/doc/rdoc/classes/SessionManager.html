195:     
196:     # default to no cookie key found:
197:     cookie_key = false
198:     
199:     # gets the cookie array from the request object
200:     cookie_raw = msg.request.cookies
201:     
202:     # checks, if a cookie named 'ses_key' is found
203:     if cookie_raw.has_key?('ses_key')
204:       
205:       # gets just the data itself (discards comment, domain, exipiry etc)
206:       cookie_key = cookie_raw['ses_key'].split(':')[0]
207:       
208:     end
209:     
210:     # if a cookie key is found (non-false), checks if it's valid
211:     if cookie_key
212:       
213:       # checks for validity by looking the key up in @session_cookie_keys
214:       cookie_key_exist = @session_cookie_keys.has_key?( cookie_key )
215:       
216:       # sets the cookie key to false, if it doesn't exist
217:       cookie_key = false unless cookie_key_exist
218:       
219:     end
220:     
221:     # at this point, the cookie key seems valid:
222:     if cookie_key and cookie_key_exist
223:       
224:       # get the session identifier
225:       ses_id = @session_cookie_keys[ cookie_key ]
226:       
227:       # get the last session key from session data
228:       ses_key = @sessions[ses_id][:ses_key]
229:       
230:       # make additional checks on the session validity (expiry etc)
231:       ses_status = check_ses( msg, ses_key )
232:       
233:       # delete the old key:
234:       @session_cookie_keys.delete( cookie_key )
235:       
236:       # status is either true (valid) or false (invalid)
237:       # based on the result of check_ses
238:       if ses_status
239:         
240:         # get a new cookie key
241:         cookie_key = @randgen.get(@config[:cookie_key_multiplier]).join('')
242:         
243:         # map the new cookie key to the old session identifier
244:         @session_cookie_keys[ cookie_key ] = ses_id
245:         
246:         # binds the new cookie key to the old session data
247:         @sessions[ses_id][:cookie_key] = cookie_key
248:         
249:         # Sets the restored_session flag of msg to true
250:         # It signals plugins to re-set data
251:         msg.restored_session = true
252:         
253:         # Sets the new_session flag of msg to false
254:         # It signals plugins to not create new server-side values
255:         msg.new_session = false
256:         
257:         # tells ValueManager to re-send client-side HValue objects
258:         # with data to the client
259:         $VALUES.resend_session_values( msg )
260:       
261:       # if the session is not valid, make sure to mark the
262:       # cookie key as invalid (false)
263:       else
264:         cookie_key = false
265:       end
266:     end
267:     
268:     # if the cookie key failed validation in the
269:     # tests above, create a new session instead
270:     unless cookie_key
271:       cookie_key = init_ses( msg )
272:       ses_status = true
273:     end
274:     
275:     # Uses a cookie comment to tell the user what the
276:     # cookie is for, change it to anything valid in the
277:     # configuration.
278:     ses_cookie_comment = @config[:ses_cookie_comment]
279:     
280:     ## mod_rewrite changes the host header to x-forwarded-host:
281:     if msg.request.header.has_key?('x-forwarded-host')
282:       domain = msg.request.header['x-forwarded-host']
283:       
284:     ## direct access just uses host (at least mongrel
285:     ## does mod_rewrite header translation):

288:     end
289:

292:     ## but not an ip-address, prepend it with

298:     ## Otherwise, use the domain as-is

300:       ses_cookie_domain  = domain

302:     
303:     ## uses the timeout to declare the max age

305:     ## it, when it expires.
306:     ses_cookie_max_age = @config[:timeout_secs]
307:     
308:     ## Only match the handshaking address of himle,

122:   def check_ses( msg, ses_key )
123:     
124:     # first, check if the session key exists (xhr)
125:     if @session_keys.has_key?( ses_key )
126:       
127:       # get the session's id based on its key
128:       ses_id   = @session_keys[ ses_key ]
129:       
130:       # get the session's data based on its id
131:       ses_data = @sessions[ ses_id ]
132:       
133:       # new time-out
134:       ses_data[:timeout] = Time.now.to_i + @config[:timeout_secs]
135:       
136:       ### extra security
137:       # re-generates the ses_key for each xhr
138:       if @config[:disposable_keys]
139:         
140:         # disposes the old (current) ses_key:
141:         @session_keys.delete( ses_key )
142:         
143:         # gets a new ses_key:
144:         ses_key = @randgen.get_one
145:         
146:         # re-maps the session id to the new key
147:         @session_keys[ses_key] = ses_id
148:         
149:         # changes the session key in the session data
150:         ses_data[:ses_key] = ses_key
151:         
152:         # tell the client what its new session key is
153:         msg.reply "HTransporter.ses_id='#{ses_key}';"
154:       end
155:       ### /extra security
156:       
157:       ### Bind the session data and id to the message object
158:       msg.session = ses_data
159:       msg.ses_id  = ses_id
160:       
161:       ## Return success
162:       return true
163:     
164:     ## The session was either faked or expired:
165:     else
166:       ### Tells the client to stop connecting with its session key and reload instead to get a new one.
167:       stop_client_with_message( msg,
168:         @config[:messages][:invalid_session][:title],
169:         @config[:messages][:invalid_session][:descr],
170:         @config[:messages][:invalid_session][:uri]
171:       )
172:       
173:       ## Return failure
174:       return false
175:     end
176:     
177:   end

338:     ## The 'ses_id' request query key is required.

343:       return false
344:     else
345:       
346:       ## get the ses_key from the request query:
347:       ses_key = request.query[ 'ses_id' ]
348:       
349:       ## The message object binds request, response
350:       ## and all user/session -related data to one
351:       ## object, which is passed around where
352:       ## request/response/user/session -related
353:       ## data is needed.
354:       msg = Message.new( request, response )
355:       
356:       ## The client tells that its ses_key is '0',
357:       ## until the server tells it otherwise.
358:       if ses_key == '0'
359:         
360:         # If Broker encounters a '/hello' request, it
361:         # sets cookies to true.
362:         #
363:         # It means that a session should have its cookies
364:         # checked.
365:         #
366:         if cookies
367:           ses_status = check_cookie( msg )
368:         # Otherwise, we just create a new session:
369:         else
370:           init_ses( msg )
371:           ses_status = true
372:         end
373:       
374:       # for non-'0' ses_keys:
375:       else
376:         
377:         ## Validate the session key
378:         ses_status = check_ses( msg, ses_key )
379:         
380:       end # /ses_id
381:       
382:       ## msg.ses_valid is false by default, meaning
383:       ## it's not valid or hasn't been initialized.
384:       msg.ses_valid = ses_status

 53:   def init_ses( msg )
 54:     
 55:     ## Assigns new timeout for the session
 56:     time_now = Time.now.to_i # seconds since epoch
 57:     timeout  = time_now + @config[:timeout_secs]
 58:     
 59:     ## Creates a new session key
 60:     ses_key    = @randgen.get_one
 61:     
 62:     ## Creates a new cookie key
 63:     cookie_key = @randgen.get(@config[:cookie_key_multiplier]).join('')
 64:     
 65:     ## Makes a new database row for the session, returns its id
 66:     ses_id     = new_ses_id( cookie_key, ses_key, timeout )
 67:     
 68:     ### Default session data structure,
 69:     ### Please don't mess with it, unless you know exactly what you are doing.
 70:     ses_data = {
 71:       
 72:       # the time, when the session will time out
 73:       :timeout    =>  timeout,
 74:       
 75:       # session id, used internally
 76:       :ses_id     =>  ses_id,
 77:       
 78:       # session key, used externally (client xhr)
 79:       :ses_key    =>  ses_key,
 80:       
 81:       # session key, used externally (client cookies)
 82:       :cookie_key =>  cookie_key,
 83:       
 84:       # user id, map to your own user management code
 85:       :user_id    =>  0,
 86:       
 87:       # valuemanager data
 88:       :values     => {
 89:         :sync  => [],  # value id's to sync to client
 90:         :check => [],  # value id's to validate in server (from client)
 91:         :by_id => {}   # values by id
 92:       }
 93:     }
 94:     
 95:     # bind the session data to @sessions by its id
 96:     @sessions[ ses_id ] = ses_data
 97:     
 98:     # map the key back to the id
 99:     @session_keys[ ses_key ] = ses_id
100:     
101:     # map the ses_id to cookie key
102:     @session_cookie_keys[ cookie_key ] = ses_id
103:     
104:     ### Tell the client what the new key is
105:     msg.reply "HTransporter.ses_id='#{ses_key}';"
106:     
107:     ### Set the session data and id to the message object
108:     msg.session = ses_data
109:     msg.ses_id  = ses_id
110:     
111:     # Flag the session as new, so associated
112:     # plugins know when to create new data
113:     msg.new_session = true
114:     
115:     # Returns the cookie key, so it can be sent in the response header
116:     return cookie_key
117:     
118:   end
