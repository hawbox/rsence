scan_dirs()

The main method to load and purge the theme cache

[Source]

     # File lib/file/filecache.rb, line 63
 63:   def scan_dirs
 64:     
 65:     # Just return, if the scan lock is on
 66:     return if @busy_scanning == true
 67:     
 68:     # Set the scan lock
 69:     @busy_scanning = true
 70:     
 71:     # Root path of the compiled js client framework
 72:     ui_path = $config[:client_parts][:js]
 73:     
 74:     # Clean hash for js data and properties
 75:     js_cache = {}
 76:     
 77:     Dir.entries( ui_path ).each do |path_item|
 78:       file_path = File.join( ui_path, path_item )
 79:       suf = suffix( file_path )
 80:       if suf == '.js'
 81:         file_key = path_item[0..-4]
 82:         js_cache[file_key] = getfile(file_path)
 83:       end
 84:     end
 85:     
 86:     # Root path of the themes
 87:     themes_path = $config[:client_parts][:themes]
 88:     
 89:     # Clean hash for theme data and properties
 90:     theme_cache = {}
 91:     
 92:     # Scans all themes
 93:     Dir.entries( themes_path ).each do |theme_name|
 94:       
 95:       # Checks, if the theme path is a directory
 96:       theme_path = File.join(themes_path,theme_name)
 97:       is_dir = File.stat(theme_path).directory?
 98:       
 99:       # Skips dirs starting with a dot
100:       if theme_name[0].chr != '.' and is_dir
101:         
102:         # initializes a theme-specific hash for css/html/gfx entries
103:         unless theme_cache.has_key?(theme_name)
104:           theme_cache[theme_name] = {'css'=>{},'html'=>{},'gfx'=>{}}
105:         end
106:         
107:         # build sub-paths for css/html/gfx entries
108:         gfx_path   = File.join(theme_path,'gfx')
109:         css_path   = File.join(theme_path,'css')
110:         html_path  = File.join(theme_path,'html')
111:         
112:         # Scans for css theme files
113:         Dir.entries( css_path ).each do |path_item|
114:           file_path = File.join( css_path, path_item )
115:           suf = suffix( file_path )
116:           if suf == '.css'
117:             theme_cache[theme_name]['css'][path_item] = getfile(file_path)
118:           end
119:         end
120:         
121:         # Scans for html theme files
122:         Dir.entries( html_path ).each do |path_item|
123:           file_path = File.join( html_path, path_item )
124:           suf = suffix( file_path )
125:           if suf == '.html'
126:             theme_cache[theme_name]['html'][path_item] = getfile(file_path)
127:           end
128:         end
129:         
130:         # Scans for gfx theme files
131:         Dir.entries( gfx_path ).each do |path_item|
132:           file_path = File.join( gfx_path, path_item )
133:           suf = suffix( file_path )
134:           if ['.jpg','.gif','.swf','.png'].include?(suf)
135:             theme_cache[theme_name]['gfx'][path_item] = getfile(file_path)
136:           end
137:         end
138:       end
139:     end
140:     
141:     # replaces the current caches with the fresh ones
142:     @js_cache = js_cache
143:     @theme_cache = theme_cache
144:     
145:     # time of last scan:
146:     @scan_time = Time.now
147:     
148:     # release lock
149:     @busy_scanning = false
150:     
151:   end
