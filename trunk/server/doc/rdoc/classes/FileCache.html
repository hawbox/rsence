scan_dirs()

The main method to load and purge the theme cache

[Source]

     # File lib/file/filecache.rb, line 56
 56:   def scan_dirs
 57:     
 58:     # Just return, if the scan lock is on
 59:     return if @busy_scanning == true
 60:     
 61:     # Set the scan lock
 62:     @busy_scanning = true
 63:     
 64:     # Root path of the compiled js client framework
 65:     ui_path = $config[:client_parts][:js]
 66:     
 67:     # Clean hash for js data and properties
 68:     js_cache = {}
 69:     
 70:     Dir.entries( ui_path ).each do |path_item|
 71:       file_path = File.join( ui_path, path_item )
 72:       suf = suffix( file_path )
 73:       if suf == '.js'
 74:         file_key = path_item[0..-4]
 75:         js_cache[file_key] = getfile(file_path)
 76:       end
 77:     end
 78:     
 79:     # Root path of the themes
 80:     themes_path = $config[:client_parts][:themes]
 81:     
 82:     # Clean hash for theme data and properties
 83:     theme_cache = {}
 84:     
 85:     # Scans all themes
 86:     Dir.entries( themes_path ).each do |theme_name|
 87:       
 88:       # Checks, if the theme path is a directory
 89:       theme_path = File.join(themes_path,theme_name)
 90:       is_dir = File.stat(theme_path).directory?
 91:       
 92:       # Skips dirs starting with a dot
 93:       if theme_name[0].chr != '.' and is_dir
 94:         
 95:         # initializes a theme-specific hash for css/html/gfx entries
 96:         unless theme_cache.has_key?(theme_name)
 97:           theme_cache[theme_name] = {'css'=>{},'html'=>{},'gfx'=>{}}
 98:         end
 99:         
100:         # build sub-paths for css/html/gfx entries
101:         gfx_path   = File.join(theme_path,'gfx')
102:         css_path   = File.join(theme_path,'css')
103:         html_path  = File.join(theme_path,'html')
104:         
105:         # Scans for css theme files
106:         Dir.entries( css_path ).each do |path_item|
107:           file_path = File.join( css_path, path_item )
108:           suf = suffix( file_path )
109:           if suf == '.css'
110:             theme_cache[theme_name]['css'][path_item] = getfile(file_path)
111:           end
112:         end
113:         
114:         # Scans for html theme files
115:         Dir.entries( html_path ).each do |path_item|
116:           file_path = File.join( html_path, path_item )
117:           suf = suffix( file_path )
118:           if suf == '.html'
119:             theme_cache[theme_name]['html'][path_item] = getfile(file_path)
120:           end
121:         end
122:         
123:         # Scans for gfx theme files
124:         Dir.entries( gfx_path ).each do |path_item|
125:           file_path = File.join( gfx_path, path_item )
126:           suf = suffix( file_path )
127:           if ['.jpg','.gif','.swf','.png'].include?(suf)
128:             theme_cache[theme_name]['gfx'][path_item] = getfile(file_path)
129:           end
130:         end
131:       end
132:     end
133:     
134:     # replaces the current caches with the fresh ones
135:     @js_cache = js_cache
136:     @theme_cache = theme_cache
137:     
138:     # time of last scan:
139:     @scan_time = Time.now
140:     
141:     # release lock
142:     @busy_scanning = false
143:     
144:   end
