     # File lib/file/ticketserve.rb, line 193
193:   def serve( msg, content, format='PNG', type=:img )
194:     
195:     # gets a new, unique identifier
196:     ticket_id = @randgen.get_one
197:     
198:     # serve image
199:     if type == :img
200:       
201:       # makes sure the format is in uppper case
202:       format.upcase!
203:       
204:       # checks, that the format is a supported image type
205:       unless @content_types.keys.include?( format )
206:         puts "ImgServe.serve: invalid format (#{format.inspect})" if $DEBUG_MODE
207:         return '/i/invalid.gif'
208:       end
209:       
210:       # changes the format to GIF for IE6
211:       format = 'GIF' if msg.ie6 and format == 'PNG'
212:       
213:       # stores the image data and meta-data ready to be served
214:       storage_hash = @imgs
215:       storage_arr = [format,0,content,msg.ses_id]
216:       
217:       # return an uri that will respond to the data
218:       uri = "/i/#{ticket_id}.#{format.downcase}"
219:     
220:     # serve file
221:     elsif type == :file
222:       
223:       # re-map content_type and filename meta-data from format
224:       (content_type, filename) = format
225:       
226:       # content size for the header
227:       content_size = content.size.to_s
228:       
229:       storage_hash = @files
230:       storage_arr = [content_type,content_size,content,msg.ses_id,filename]
231:       
232:       uri = "/f/#{ticket_id}"
233:     end
234:     
235:     # makes sure, that the storage array has a sub-array for sessions (to aid session-based cleanup)
236:     storage_hash[:ses_ids][msg.ses_id] = [] unless @imgs[:ses_ids].has_key?(msg.ses_id)
237:     
238:     # adds the id to the session-specific storage
239:     storage_hash[:ses_ids][msg.ses_id].push( ticket_id )
240:     
241:     storage_hash[:by_id][ticket_id] = storage_arr
242:     
243:     return uri
244:   end
