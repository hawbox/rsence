     # File plugins/main/main.rb, line 111
111:   def idle(msg)
112:     
113:     mses = msg.session[:main]
114:     
115:     ## Prioritizes segmented execution to prevent
116:     ## the client from choking on itself
117:     if mses[:boot] == 0
118:       
119:       ## js/start.js includes the client's initial settings
120:       msg.reply require_js_once(msg,'start')
121:       
122:       ## url_responder is bound in the client-space
123:       ## to tell the server its status by updating its value
124:       msg.reply require_js_once(msg,'url_responder')
125:       msg.reply "urlCatcher = new URLCatcher('#{mses[:location_href].val_id}');"
126:     
127:     ## Second stage enables SesWatcher that changes :client_time every 60 seconds.
128:     ## Causes the client to poll the server on regular intervals, when polling mode
129:     ## is disabled.
130:     elsif mses[:boot] == 1
131:       
132:       # 60000ms = 60secs
133:       msg.reply "sesWatcher = new SesWatcher(60000,'#{mses[:client_time].val_id}');"
134:     
135:     ## Processes delayed calls, if the
136:     ## :delayed_calls -array contains something to process.
137:     elsif not mses[:delayed_calls].empty?
138:       
139:       ## Limits the amount of delayed calls to process to 4.
140:       ## Prevents the client from choking even when the server
141:       ## load is light.
142:       if mses[:delayed_calls].size < 4
143:         call_count = mses[:delayed_calls].size
144:       else
145:         call_count = 4
146:       end
147:       
148:       time_start = Time.now.to_f
149:       time_taken = 0.0
150:       
151:       ## process delayed calls, until:
152:       ## - over 200ms of cpu time has been spent
153:       ## - the :delayed_calls -array is empty
154:       ## - call_count limit is reached
155:       until time_taken > 0.2 or mses[:delayed_calls].size == 0 or call_count == 0
156:         
157:         # gets the next call
158:         delayed_call = mses[:delayed_calls].shift
159:         if $DEBUG_MODE
160:           puts '-='*30
161:           puts "delayed_call: #{delayed_call.inspect}"
162:           puts '=-'*30
163:         end
164:         
165:         # strings are always javascript, used for segmenting client load
166:         if delayed_call.class == String
167:           msg.reply delayed_call
168:         
169:         # arrays are plugin calls
170:         elsif delayed_call.class == Array
171:           
172:           # ['plugin_name', 'method_name'] pairs call the named plugin:method with just msg
173:           if delayed_call.size == 2
174:             (plugin_name,method_name) = delayed_call
175:             msg.run(plugin_name,method_name,msg)
176:           
177:           # if the array contains more items, they are used as additional method params
178:           else
179:             (plugin_name,method_name) = delayed_call[0..1]
180:             method_params = delayed_call[2..-1]
181:             msg.run(plugin_name,method_name,msg,*method_params)
182:           end
183:           
184:         end
185:         
186:         ## calculates time taken
187:         time_taken = Time.now.to_f - time_start
188:         
189:         call_count -= 1
190:       end
191:       
192:       ## Sets the client into poll mode, unless the :delayed_calls -array is empty
193:       if mses[:boot] > 3
194:         if mses[:delayed_calls].empty?
195:           msg.reply( "document.title = #{$config[:indexhtml_conf][:loaded_title].inspect};" )
196:         end
197:         msg.reply( "HTransporter.setPollMode(#{(not mses[:delayed_calls].empty?).inspect});" )
198:       end
199:       
200:     ## When nothing is delayed and the fifth poll has been made,
201:     ## sets the client to non-polling-mode, having only HValue
202:     ## changes trigger new requests.
203:     elsif mses[:boot] > 3
204:       msg.reply "HTransporter.setPollMode(false);"
205:       msg.reply("document.title = #{$config[:indexhtml_conf][:loaded_title].inspect};")
206:     end
207:     
208:     ## Increment the counter forever.
209:     mses[:boot] += 1
210:     
211:   end

url_responder(msg,location_href)

url_responder gets called whenever the page location.href changes, enabled virtual uris to enable back/forward/bookmarking in browsers, when software is coded to support it.

Client-side support is included in js/url_responder.js

Also allows virtual-host -like behavior, if software is coded to support it.

[Source]

    # File plugins/main/main.rb, line 34
34:   def url_responder(msg,location_href)
35:     
36:     mses = msg.session[:main]
37:     
38:     # Virtual locations:
39:     if location_href.data.include?('#')
40:       
41:       # split 'http://localhost:8001/#/some_uri'
42:       #   -> ['http://localhost:8001/','/some_uri']
43:       mses[:url] = location_href.data.split('#')
44:       
45:       virtual_uri = mses[:url][1]
46:       
47:       # built-in support for signing out, deletes the
48:       # server-side session and reloads the page
49:       if virtual_uri == '/sign_out'
50:         msg.reply( "location.href='/?';" )
51:         msg.expire_session()
52:       end
53:       
54:     else
55:       
56:       # assigns only a virtual url
57:       mses[:url] = [location_href.data]
58:       
59:     end
60:     
61:     # url_responder always accepts locations
62:     return true
63:     
64:   end
