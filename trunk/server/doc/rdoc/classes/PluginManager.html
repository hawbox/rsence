 59:   def scan
 60:     
 61:     # loop through all plugin-mainlevel directories
 62:     @dirs.each do |plugin_dir|
 63:       
 64:       # makes sure the plugin dir is a dir
 65:       # (there might be files like readme.txt or licence.txt)
 66:       if(FileTest.directory?(plugin_dir))
 67:         
 68:         # goes through all plugins in the
 69:         # plugin dir in alphabetical order:
 70:         Dir.new(plugin_dir).sort.each do |plugin_name|
 71:           
 72:           # skip, if the plugin starts with a dot, like '.svn', '.' and '..'
 73:           next if(plugin_name =~ /^\./)
 74:           
 75:           # sets the plugin path
 76:           @@curr_plugin_path = File.join(plugin_dir,plugin_name)
 77:           
 78:           # checks that the plugin is a dir
 79:           next unless(FileTest.directory?(@@curr_plugin_path))
 80:           
 81:           # expects to find a 'plugin_dir/plugin_name/plugin_name.rb' file
 82:           filename = File.join(@@curr_plugin_path,"#{plugin_name}.rb")
 83:           next unless File.exist?(filename)
 84:           
 85:           # if the plugin contains a 'disabled' flag-file, it skips to the next
 86:           next if File.exist?(File.join(@@curr_plugin_path,'disabled'))
 87:           
 88:           # Create a new, anonymous module as the plugin namespace.
 89:           plugin_module = Module.new
 90:           
 91:           ##### Evaluates the plugin as a string in an anonymous module
 92:           begin
 93:             plugin_as_string = IO.readlines( filename ).join
 94:             plugin_module.module_eval( plugin_as_string )
 95:             
 96:           ###### If initialization fails, print a nice stack trace
 97:           rescue => e
 98:             puts "=="*40 if $DEBUG_MODE
 99:             puts "WARN: Plugin #{plugin_name} failed to initialize."
100:             if $DEBUG_MODE
101:               puts "--"*40
102:               puts e.message.gsub('(eval):',%{#{File.join(@@curr_plugin_path,"#{plugin_name}.rb")}:})
103:               puts "  #{e.backtrace.join("\n  ")}"
104:               puts "=="*40
105:             end
106:           end
107:           
108:         end
109:         
110:       end
111:       
112:     end

117:     
118:   end
