 51:   def scan
 52:     
 53:     # loop through all plugin-mainlevel directories
 54:     @dirs.each do |plugin_dir|
 55:       
 56:       # makes sure the plugin dir is a dir
 57:       # (there might be files like readme.txt or licence.txt)
 58:       if(FileTest.directory?(plugin_dir))
 59:         
 60:         # goes through all plugins in the
 61:         # plugin dir in alphabetical order:
 62:         Dir.new(plugin_dir).sort.each do |plugin_name|
 63:           
 64:           # skip, if the plugin starts with a dot, like '.svn', '.' and '..'
 65:           next if(plugin_name =~ /^\./)
 66:           
 67:           # sets the plugin path
 68:           @@curr_plugin_path = File.join(plugin_dir,plugin_name)
 69:           
 70:           # checks that the plugin is a dir
 71:           next unless(FileTest.directory?(@@curr_plugin_path))
 72:           
 73:           # expects to find a 'plugin_dir/plugin_name/plugin_name.rb' file
 74:           filename = File.join(@@curr_plugin_path,"#{plugin_name}.rb")
 75:           next unless File.exist?(filename)
 76:           
 77:           # if the plugin contains a 'disabled' flag-file, it skips to the next
 78:           next if File.exist?(File.join(@@curr_plugin_path,'disabled'))
 79:           
 80:           # Create a new, anonymous module as the plugin namespace.
 81:           plugin_module = Module.new
 82:           
 83:           ##### Evaluates the plugin as a string in an anonymous module
 84:           begin
 85:             plugin_as_string = IO.readlines( filename ).join
 86:             plugin_module.module_eval( plugin_as_string )
 87:             
 88:           ###### If initialization fails, print a nice stack trace
 89:           rescue => e
 90:             puts "=="*40 if DEBUG_MODE
 91:             puts "WARN: Plugin #{plugin_name} failed to initialize."
 92:             if DEBUG_MODE
 93:               puts "--"*40
 94:               puts e.message.gsub('(eval):',%{#{File.join(@@curr_plugin_path,"#{plugin_name}.rb")}:})
 95:               puts "  #{e.backtrace.join("\n  ")}"
 96:               puts "=="*40
 97:             end
 98:           end
 99:           
100:         end
101:         
102:       end
103:       
104:     end
105:     
106:     @@curr_plugin_path = nil
107:     
108:   end
