    # File examples/plugins/demo_app/demo_app.rb, line 85
85:   def calc_clock_coords( step, steps_max, distance_from_center )
86:     angle_step_degrees = (360.0/steps_max).round
87:     angle = ((step*angle_step_degrees-180.0) * Math::PI) / 180.0
88:     x = (0-Math.sin(angle)*distance_from_center).round + @clock_center_x
89:     y = (0+Math.cos(angle)*distance_from_center).round + @clock_center_y
90:     return [x,y]
91:   end

render_clock_bg()

Renders the clock background, it‘s shared amongst sessions and it‘s rendered once only, when the application starts. It‘s called from ClockApp::open()

[Source]

     # File examples/plugins/demo_app/demo_app.rb, line 96
 96:   def render_clock_bg
 97:     
 98:     # Create the image object
 99:     clock_bg = Magick::Image.new( @clock_width, @clock_height ) { 
100:       self.background_color = 'transparent'
101:     }
102:     
103:     # Create the drawing command object
104:     draw_bg = Magick::Draw.new
105:     
106:     # Set styles for circle:
107:     draw_bg.fill( @background_color )
108:     draw_bg.fill_opacity( @background_opacity )
109:     draw_bg.stroke( @border_color )
110:     draw_bg.stroke_width( @border_thickness )
111:     draw_bg.stroke_opacity( @border_opacity )
112:     
113:     # Draw circle:
114:     draw_bg.circle(
115:       @clock_center_x, @clock_center_y,    # circle center point
116:       @border_thickness/2, @clock_center_y # circle left edge point
117:     )
118:     
119:     # Set styles for hour dots:
120:     draw_bg.fill( @hours_dot_color )
121:     draw_bg.fill_opacity( @hours_dot_opacity )
122:     draw_bg.stroke( 'transparent' ) # no stroke
123:     draw_bg.stroke_width( 0 )       # no stroke
124:     
125:     # Draw circles for each hour
126:     12.times do |hour|
127:       (x,y) = calc_clock_coords( hour, 12, @hours_dot_position )
128:       (x2,y2) = calc_clock_coords( hour, 12, @hours_dot_position + @hours_dot_size )
129:       draw_bg.circle(x,y,x2,y2)
130:     end
131:     
132:     # Render the drawing commands to pixels:
133:     draw_bg.draw( clock_bg )
134:     
135:     # Convert the raw pixels to png-image data
136:     clock_img_data = clock_bg.to_blob{ self.format='PNG' }
137:     
138:     # Send the image to the TICKETSERVE cache as a 'resource', basically
139:     # meaning that it should be stored forever or until it's manually removed.
140:     # It returns a generated uri stored as @clock_bg_img_url
141:     @clock_bg_img_url = $TICKETSERVE.serve_rsrc( clock_img_data, 'image/png' )
142:     
143:   end
render_clock_fg(msg, time_now=false)

Renders the clock foreground, it‘s a disposable image sent only once to the client. It takes an optional parameter; time specified in seconds since epoch

[Source]

     # File examples/plugins/demo_app/demo_app.rb, line 150
150:   def render_clock_fg(msg, time_now=false)
151:     
152:     # Create a date/time object based on the argument given 
153:     if time_now == false
154:       time_now = Time.now
155:     else
156:       time_now = Time.at( time_now )
157:     end
158:     
159:     # Get the time in 12.(minutes*1.66) -format
160:     hour = time_now.hour
161:     hour -= 12 if hour >= 12
162:     minutes = time_now.min
163:     hour += (minutes/60.0)
164:     
165:     # Create the image object
166:     clock_fg = Magick::Image.new( @clock_width, @clock_height ) { 
167:       self.background_color = 'transparent'
168:     }
169:     
170:     # Create the drawing command object
171:     draw_fg = Magick::Draw.new
172:     
173:     # Set clock arm style
174:     draw_fg.stroke_opacity( @clock_arms_opacity )
175:     
176:     draw_fg.stroke( @hour_arm_color )
177:     draw_fg.stroke_width( @hour_arm_width )
178:     
179:     # Draw the hour arm:
180:     (hour_x,hour_y) = calc_clock_coords(hour,12,@hour_arm_length)
181:     draw_fg.line(@clock_center_x,@clock_center_y,hour_x,hour_y)
182:     
183:     # Set the minute arm style:
184:     draw_fg.stroke( @minute_arm_color )
185:     draw_fg.stroke_width( @minute_arm_width )
186:     
187:     # Draw the minute arm:
188:     (minutes_x,minutes_y) = calc_clock_coords(minutes,60,@minute_arm_length)
189:     draw_fg.line(@clock_center_x,@clock_center_y,minutes_x,minutes_y)
190:     
191:     # Render the drawing commands to pixels:
192:     draw_fg.draw( clock_fg )
193:     
194:     # Store the image as a disposable image object, it expires automatically.
195:     clock_fg_img_url = msg.serve_img(clock_fg,'PNG')
196:     
197:     # Return the image uri returned by the image cache
198:     return clock_fg_img_url
199:   end
