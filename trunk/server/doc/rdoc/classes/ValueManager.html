 55:   def resend_session_values( msg )
 56:     
 57:     # with disposable keys enabled,
 58:     # sessions restored from cookie_key replace
 59:     # all val_id:s with new ones
 60:     if @config[:disposable_keys]
 61:       
 62:       # get the session values
 63:       ses_values = msg.session[:values]
 64:       
 65:       # loop through the value id:s
 66:       old_ids = ses_values[:by_id].keys
 67:       old_ids.each do |old_id|
 68:         
 69:         # make a new id
 70:         new_id = @randgen.get_one
 71:         
 72:         # get the hvalue
 73:         val_obj = ses_values[:by_id][old_id]
 74:         
 75:         # replace the old id in the hvalue itself
 76:         val_obj.val_id = new_id
 77:         
 78:         # re-associate the value with the new id
 79:         ses_values[:by_id][new_id] = val_obj
 80:         ses_values[:by_id].delete(old_id)
 81:         
 82:         # replace the id in the unvalidated values (:check) array
 83:         if ses_values[:check].include?(old_id)
 84:           old_idx = ses_values[:check].index(old_id)
 85:           ses_values[:check][old_idx] = new_id
 86:         end
 87:         
 88:         # replace the id in the unsynchronized values (:sync) array
 89:         if ses_values[:sync].include?(old_id)
 90:           old_idx = ses_values[:sync].index(old_id)
 91:           ses_values[:sync][old_idx] = new_id
 92:         end
 93:         
 94:         # tell the hvalue to send its client-side initialization
 95:         val_obj.restore( msg )
 96:         
 97:       end
 98:     
 99:     ## keeping the id:s between page reloads is faster, but less tamper-proof
100:     else
101:       msg.session[:values][:by_id].each_key do |val_id|
102:         msg.session[:values][:by_id][val_id].restore( msg )
103:       end
104:     end
