 63:   def resend_session_values( msg )
 64:     
 65:     # with disposable keys enabled,
 66:     # sessions restored from cookie_key replace
 67:     # all val_id:s with new ones
 68:     if @config[:disposable_keys]
 69:       
 70:       # get the session values
 71:       ses_values = msg.session[:values]
 72:       
 73:       # loop through the value id:s
 74:       old_ids = ses_values[:by_id].keys
 75:       old_ids.each do |old_id|
 76:         
 77:         # make a new id
 78:         new_id = @randgen.get_one
 79:         
 80:         # get the hvalue
 81:         val_obj = ses_values[:by_id][old_id]
 82:         
 83:         # replace the old id in the hvalue itself
 84:         val_obj.val_id = new_id
 85:         
 86:         # re-associate the value with the new id
 87:         ses_values[:by_id][new_id] = val_obj
 88:         ses_values[:by_id].delete(old_id)
 89:         
 90:         # replace the id in the unvalidated values (:check) array
 91:         if ses_values[:check].include?(old_id)
 92:           old_idx = ses_values[:check].index(old_id)
 93:           ses_values[:check][old_idx] = new_id
 94:         end
 95:         
 96:         # replace the id in the unsynchronized values (:sync) array
 97:         if ses_values[:sync].include?(old_id)
 98:           old_idx = ses_values[:sync].index(old_id)
 99:           ses_values[:sync][old_idx] = new_id
100:         end
101:         
102:         # tell the hvalue to send its client-side initialization
103:         val_obj.restore( msg )
104:         
105:       end
106:     
107:     ## keeping the id:s between page reloads is faster, but less tamper-proof
108:     else
109:       msg.session[:values][:by_id].each_key do |val_id|
110:         msg.session[:values][:by_id][val_id].restore( msg )
111:       end
112:     end
