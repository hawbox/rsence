     # File lib/http/broker.rb, line 80
 80:   def get
 81:     
 82:     ## The full request URI:
 83:     uri = @request.fullpath
 84:     
 85:     ## /j processes client framework files (js & themes)
 86:     if uri[0..2] == '/H/'
 87:       puts "/H: #{uri.inspect}" if $DEBUG_MODE
 88:       $FILESERVE.get( @request, @response )
 89:     
 90:     ## /i returns disposable RMagick objects rendered to data
 91:     elsif uri[0..2] == '/i/'
 92:       puts "/i: #{uri.inspect}" if $DEBUG_MODE
 93:       $TICKETSERVE.get( @request, @response, :img )
 94:     
 95:     ## /d returns static data resources
 96:     elsif uri[0..2] == '/d/'
 97:       puts "/d: #{uri.inspect}" if $DEBUG_MODE
 98:       $TICKETSERVE.get( @request, @response, :rsrc )
 99:     
100:     ## /f return disposable data resources
101:     elsif uri[0..2] == '/f/'
102:       puts "/f: #{uri.inspect}" if $DEBUG_MODE
103:       $TICKETSERVE.get( @request, @response, :file )
104:     
105:     ## /B return smart data objects
106:     elsif uri[0..2] == '/b/'
107:       puts "/b: #{uri.inspect}" if $DEBUG_MODE
108:       $TICKETSERVE.get( @request, @response, :blobobj )
109:     
110:     ## special case for favicon
111:     elsif uri == '/favicon.ico'
112:       $TICKETSERVE.favicon( @request, @response )
113:     
114:     ## empty html page for the uploader iframe
115:     elsif uri == '/U/iframe_html'
116:       puts "/U/iframe_html: #{uri.inspect}" if $DEBUG_MODE
117:       @response.status = 200
118:       http_body = '<html><head><title>Empty Iframe for Uploading</title></head><body></body></html>'
119:       @response['content-type'] = 'text/html; charset=UTF-8'
120:       @response['content-size'] = http_body.size.to_s
121:       @response.body = http_body
122:     
123:     ## default index html page
124:     elsif uri == '/'
125:       puts "/: #{uri.inspect}" if $DEBUG_MODE
126:       $INDEXHTML.get( @request, @response )
127:     
128:     ## all other get -requests load the 404 html page 
129:     else
130:       puts "/404: #{uri.inspect}" if $DEBUG_MODE
131:       @response.status = 404
132:       err404 = '<html><head><title>404 - Page Not Found</title></head><body>404 - Page Not Found</body></html>'
133:       @response['content-type'] = 'text/html; charset=UTF-8'
134:       @response['content-size'] = err404.size.to_s
135:       @response.body = err404
136:     end

41:   def post
42:     
43:     ## The full request URI:
44:     uri = @request.fullpath
45:     
46:     ## /x handles xhr without cookies
47:     if uri == '/x'
48:       puts "/x: #{uri.inspect}" if $DEBUG_MODE
49:       $TRANSPORTER.xhr( @request, @response, false )
50:     
51:     ## /hello handles the first xhr (with cookies, for session key)
52:     elsif uri == '/hello'
53:       puts "/hello: #{uri.inspect}" if $DEBUG_MODE
54:       $TRANSPORTER.xhr( @request, @response, true )
55:     
56:     ## /SOAP handles SOAP Requests (only when debugging, for now...)
57:     elsif uri == '/SOAP' and $DEBUG_MODE
58:       puts "/SOAP: #{uri.inspect}" if $DEBUG_MODE
59:       $TRANSPORTER.soap( @request, @response )
60:     
61:     ## /up handles the uploads inited for ticketserve
62:     ## the second part of the uri contains the disposable
63:     ## upload key that has a server-side mapping to the
64:     ## user's session
65:     elsif uri[0..2] == '/U/'
66:       puts "/U: #{uri.inspect}" if $DEBUG_MODE
67:       $TICKETSERVE.upload( @request, @response )
68:     else
69:       puts "/404: #{uri.inspect}" if $DEBUG_MODE
70:       @response.status = 404
71:       err404 = '<html><head><title>404 - Page Not Found</title></head><body>404 - Page Not Found</body></html>'
72:       @response['content-type'] = 'text/html; charset=UTF-8'
73:       @response['content-size'] = err404.size.to_s
74:       @response.body = err404
75:     end
76:     
77:   end
