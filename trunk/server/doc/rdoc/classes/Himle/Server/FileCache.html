scan_dirs()

The main method to load and purge the theme cache

[Source]

     # File lib/file/filecache.rb, line 67
 67:   def scan_dirs
 68:     
 69:     # Just return, if the scan lock is on
 70:     return if @busy_scanning == true
 71:     
 72:     # Set the scan lock
 73:     @busy_scanning = true
 74:     
 75:     # Root path of the compiled js client framework
 76:     ui_path = $config[:client_parts][:js]
 77:     
 78:     # Clean hash for js data and properties
 79:     js_cache = {}
 80:     gz_cache = {}
 81:     
 82:     Dir.entries( ui_path ).each do |path_item|
 83:       file_path = File.join( ui_path, path_item )
 84:       suf = suffix( file_path )
 85:       if suf == '.js'
 86:         file_key = path_item[0..-4]
 87:         js_cache[file_key] = getfile(file_path)
 88:       elsif suf == '.gz'
 89:         file_key = path_item[0..-4]
 90:         gz_cache[file_key] = getfile(file_path)
 91:       end
 92:     end
 93:     
 94:     # Root path of the themes
 95:     themes_path = $config[:client_parts][:themes]
 96:     
 97:     # Clean hash for theme data and properties
 98:     theme_cache = {}
 99:     
100:     # Scans all themes
101:     Dir.entries( themes_path ).each do |theme_name|
102:       
103:       # Checks, if the theme path is a directory
104:       theme_path = File.join(themes_path,theme_name)
105:       is_dir = File.stat(theme_path).directory?
106:       
107:       # Skips dirs starting with a dot
108:       if theme_name[0].chr != '.' and is_dir
109:         
110:         # initializes a theme-specific hash for css/html/gfx entries
111:         unless theme_cache.has_key?(theme_name)
112:           theme_cache[theme_name] = {'css'=>{},'html'=>{},'gfx'=>{}}
113:         end
114:         
115:         # build sub-paths for css/html/gfx entries
116:         gfx_path   = File.join(theme_path,'gfx')
117:         css_path   = File.join(theme_path,'css')
118:         html_path  = File.join(theme_path,'html')
119:         
120:         # Scans for css theme files
121:         Dir.entries( css_path ).each do |path_item|
122:           file_path = File.join( css_path, path_item )
123:           suf = suffix( file_path )
124:           if suf == '.css' or suf == '.gz'
125:             theme_cache[theme_name]['css'][path_item] = getfile(file_path)
126:           end
127:         end
128:         
129:         # Scans for html theme files
130:         Dir.entries( html_path ).each do |path_item|
131:           file_path = File.join( html_path, path_item )
132:           suf = suffix( file_path )
133:           if suf == '.html' or suf == '.gz'
134:             theme_cache[theme_name]['html'][path_item] = getfile(file_path)
135:           end
136:         end
137:         
138:         # Scans for gfx theme files
139:         Dir.entries( gfx_path ).each do |path_item|
140:           file_path = File.join( gfx_path, path_item )
141:           suf = suffix( file_path )
142:           if ['.jpg','.gif','.swf','.png'].include?(suf)
143:             theme_cache[theme_name]['gfx'][path_item] = getfile(file_path)
144:           end
145:         end
146:       end
147:     end
148:     
149:     # replaces the current caches with the fresh ones
150:     @js_cache = js_cache
151:     @gz_cache = gz_cache
152:     @theme_cache = theme_cache
153:     
154:     # time of last scan:
155:     @scan_time = Time.now
156:     
157:     # release lock
158:     @busy_scanning = false
159:     
160:   end
