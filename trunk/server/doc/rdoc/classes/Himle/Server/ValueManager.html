 68:   def resend_session_values( msg )
 69:     
 70:     # with disposable keys enabled,
 71:     # sessions restored from cookie_key replace
 72:     # all val_id:s with new ones
 73:     if @config[:disposable_keys]
 74:       
 75:       # get the session values
 76:       ses_values = msg.session[:values]
 77:       
 78:       # loop through the value id:s
 79:       old_ids = ses_values[:by_id].keys
 80:       old_ids.each do |old_id|
 81:         
 82:         # make a new id
 83:         new_id = @randgen.get_one
 84:         
 85:         # get the hvalue
 86:         val_obj = ses_values[:by_id][old_id]
 87:         
 88:         # replace the old id in the hvalue itself
 89:         val_obj.val_id = new_id
 90:         
 91:         # re-associate the value with the new id
 92:         ses_values[:by_id][new_id] = val_obj
 93:         ses_values[:by_id].delete(old_id)
 94:         
 95:         # replace the id in the unvalidated values (:check) array
 96:         if ses_values[:check].include?(old_id)
 97:           old_idx = ses_values[:check].index(old_id)
 98:           ses_values[:check][old_idx] = new_id
 99:         end
100:         
101:         # replace the id in the unsynchronized values (:sync) array
102:         if ses_values[:sync].include?(old_id)
103:           old_idx = ses_values[:sync].index(old_id)
104:           ses_values[:sync][old_idx] = new_id
105:         end
106:         
107:         # tell the hvalue to send its client-side initialization
108:         val_obj.restore( msg )
109:         
110:       end
111:     
112:     ## keeping the id:s between page reloads is faster, but less tamper-proof
113:     else
114:       msg.session[:values][:by_id].each_key do |val_id|
115:         msg.session[:values][:by_id][val_id].restore( msg )
116:       end
